│   ├── build/             # Build scripts
│   ├── fix/               # Fix scripts (archive)
│   ├── game/              # Game mechanics
│   ├── export/            # Export scripts
│   └── utils/             # Utilities
│
├── assets/                 # Static assets
│   ├── cards/             # Card images (SVG)

│   └── card-back.svg      # Card back

│
├── chain-game/             # Chain game docs
